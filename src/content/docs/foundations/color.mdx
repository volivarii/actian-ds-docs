export function flattenColorFamily(family, prefix) {
  // Flatten a DTCG color family (e.g. tokens.color.primary) into table rows.
  // Token name = "--zen-color-<prefix>-<dotted-key-path>".
  // Leaves are { $type: "color", $value: "#hex", ... }.
  const rows = [];
  function walk(obj, parts) {
    for (const [k, v] of Object.entries(obj)) {
      if (v && typeof v === "object" && "$value" in v) {
        const name = "--zen-color-" + [prefix, ...parts, k].join("-");
        rows.push({ Token: name, Value: String(v.$value) });
      } else if (v && typeof v === "object") {
        walk(v, [...parts, k]);
      }
    }
  }
  walk(family, []);
  return rows;
}
